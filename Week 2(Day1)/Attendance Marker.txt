CLASS Attendance
    ATTRIBUTE records[students][days]

    METHOD markAttendance(studentID, day, status)
        records[studentID][day] = status
    END METHOD

    METHOD updateAttendance(studentID, day, status)
        records[studentID][day] = status
    END METHOD

    METHOD generateReport(studentID)
        countPresent = 0
        FOR day = 1 TO totalDays
            IF records[studentID][day] == "P" THEN
                countPresent = countPresent + 1
            END IF
        END FOR
        PRINT "Student " + studentID + " was present " + countPresent + " days."
    END METHOD
END CLASS


FUNCTION MAIN
    CREATE system AS Attendance(30 students, 31 days)

    CALL system.markAttendance(1, 1, "P")
    CALL system.markAttendance(1, 2, "P")
    CALL system.markAttendance(1, 3, "A")
    CALL system.markAttendance(1, 4, "P")

    CALL system.updateAttendance(1, 3, "P")   // correction

    CALL system.generateReport(1)
END FUNCTION
